from selenium import webdriver
from selenium.webdriver.common.by import By
import time

def test_form_submission():
    driver = webdriver.Chrome()
    driver.get("https://formy-project.herokuapp.com/form")

    driver.find_element(By.ID, "first-name").send_keys("Gonçalo")
    driver.find_element(By.ID, "last-name").send_keys("Silva")
    driver.find_element(By.ID, "job-title").send_keys("QA Engineer")
    driver.find_element(By.ID, "radio-button-1").click()  # Education
    driver.find_element(By.ID, "checkbox-1").click()  # Gender

    driver.find_element(By.CSS_SELECTOR, ".btn.btn-lg.btn-primary").click()
    time.sleep(2)

    # Verifica mensagem de sucesso
    alert = driver.find_element(By.CLASS_NAME, "alert")
    assert "The form was successfully submitted!" in alert.text
    print("[PASSOU] Formulário submetido com sucesso")

    driver.quit()

if __name__ == "__main__":
    test_form_submission()
